(ns yvern.scramble.pages
  (:require [hiccup.core :refer [html]]))
(defn load-app [] (slurp "src/yvern/scramble/app.cljs"))
(defmacro app' []
    (if *compile-files*
      `(let [app# ~(load-app)] (fn [] app#))
      load-app))
(def app (app'))
